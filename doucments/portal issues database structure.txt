CREATE TABLE `tbl_pi_categories` (
 `pi_category_id` int(11) NOT NULL AUTO_INCREMENT,
 `pi_category_name` varchar(100) NOT NULL,
 `created_by` int(11) NOT NULL,
 `created_at` datetime NOT NULL DEFAULT current_timestamp(),
 `status` varchar(20) NOT NULL DEFAULT 'Active',
 PRIMARY KEY (`pi_category_id`),
 KEY `created_by` (`created_by`),
 CONSTRAINT `tbl_pi_categories_ibfk_1` FOREIGN KEY (`created_by`) REFERENCES `employees` (`emp_id`) ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=8 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci


CREATE TABLE `tbl_pi_reported` (
 `portal_issue_id` int(11) NOT NULL AUTO_INCREMENT,
 `pi_category_id` int(11) NOT NULL,
 `pi_category` varchar(100) NOT NULL,
 `subject` varchar(255) NOT NULL,
 `description` longtext NOT NULL,
 `issue_date` date NOT NULL,
 `priority` varchar(10) NOT NULL DEFAULT 'Low',
 `status` varchar(20) NOT NULL DEFAULT 'Pending',
 `requested_by` int(11) NOT NULL,
 `requested_at` timestamp NOT NULL DEFAULT current_timestamp(),
 `support_by` int(11) DEFAULT NULL,
 `support_at` datetime DEFAULT NULL,
 `support_comments` varchar(255) DEFAULT NULL,
 `last_edit_by` int(11) DEFAULT NULL,
 `last_edit_at` datetime DEFAULT NULL,
 PRIMARY KEY (`portal_issue_id`),
 KEY `pi_category_id` (`pi_category_id`),
 KEY `last_edit_by` (`last_edit_by`),
 KEY `requested_by` (`requested_by`),
 KEY `support_by` (`support_by`),
 CONSTRAINT `tbl_pi_reported_ibfk_1` FOREIGN KEY (`pi_category_id`) REFERENCES `tbl_pi_categories` (`pi_category_id`) ON UPDATE CASCADE,
 CONSTRAINT `tbl_pi_reported_ibfk_2` FOREIGN KEY (`last_edit_by`) REFERENCES `seaboard`.`employees` (`emp_id`) ON UPDATE CASCADE,
 CONSTRAINT `tbl_pi_reported_ibfk_3` FOREIGN KEY (`requested_by`) REFERENCES `seaboard`.`employees` (`emp_id`) ON UPDATE CASCADE,
 CONSTRAINT `tbl_pi_reported_ibfk_4` FOREIGN KEY (`support_by`) REFERENCES `seaboard`.`employees` (`emp_id`) ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci
